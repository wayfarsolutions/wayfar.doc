applyFilters({String? sortKey,List<double>? priceRange, List<double>? distanceRange, List<Service>? attributes}) {
  var newState = ParkingFiltersReadyState(allAttributes);
  newState.filterText = filterText;
  newState.sortKey = sortKey ?? "price";
  if (priceRange!=null) {
    newState.priceRange = RangeValues(priceRange[0], priceRange[1]);
  } else {
    newState.priceRange = RangeValues(0, 100);
  }
  if (distanceRange!=null) {
    newState.distanceRange = RangeValues(distanceRange[0], distanceRange[1]);
  }
  else {
    newState.distanceRange = RangeValues(0, 100);
  }
  newState.attributes = attributes ?? [];
  return newState;
}
