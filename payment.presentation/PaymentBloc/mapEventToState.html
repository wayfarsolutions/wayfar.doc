Must be implemented when a class extends Bloc. mapEventToState is called whenever an event is added and is responsible for converting that event into a new state. mapEventToState can yield zero, one, or multiple states for an event.

@override
Stream<PaymentState> mapEventToState(PaymentEvent event,) async* {
  String error = '';
  if(event is PaymentStartEvent) {
    yield PaymentLoadingState();
    final parkingFeeResult = await _getParkingFeeByIdUseCase.execute(
        GetParkingFeeByIdParams(parkingId: parkingId!));
    final parkingDiscountResult = await _getParkingDiscountByIdUseCase.execute(
        GetParkingDiscountByIdParams(parkingId: parkingId!));
    PaymentState stateR = parkingFeeResult.fold(
          (error) => PaymentErrorState(error.message),
          (parkingFee) {
            return parkingDiscountResult.fold(
                    (error) => PaymentErrorState(error.message),
                    (parkingDiscount) => PaymentProceedState(parkingPrice: parkingFee, discount: parkingDiscount));
          });
    yield stateR;

  }else if (event is PaymentAddPromoCodeEvent) {
    //TODO you can check promo code validity
    final promoCodeDiscountResult = await _getDiscountByPromoCodeUseCase.execute(
        GetDiscountByPromoCodeParams(parkingId: parkingId, promoCode: event.promoCode));
    yield promoCodeDiscountResult.fold(
          (error) => PaymentErrorState(error.message),
          (promoCodeDiscount) => (state as PaymentProceedState).copyWith(discount: promoCodeDiscount),
    );
  }
  else if (event is PaymentChangeTimeEvent) {
    yield (state as PaymentProceedState).copyWith(changePrice: true, parkingPeriodInMinutes: event.parkingPeriodInMinutes);

  } else if (event is PaymentExecuteEvent){
    if (state is PaymentProceedState){

      yield PaymentLoadingState();

      final bookingResult = await _bookParkingSpotUseCase.execute( BookParkingSpotParams(
        totalPrice: event.totalPrice,
          paymentMethodType: event.paymentMethodType,
          parkingName: event.parkingName,
          parkingAddress: event.parkingAddress,
          parkingId: event.parkingId,
        parkingStartTime: event.parkingStartTime,
        parkingEndTime: event.parkingEndTime,
      ));
      //print("Payment success!!$parkingId");

      yield bookingResult.fold(
            (error) => PaymentErrorState(error.message),
            (booking) =>  PaymentSuccessState(booking: booking),
      );
    }else if(state is PaymentSuccessState){
      yield state;
    }
  } else yield state;
}
