@override
Future<Either<Failure, AuthResult>> execute(RegisterParams params) async {
  try {
    var token = await _repository.signUp(fullName: params.fullName, username: params.username, password: params.password);
    var user = await _repository.getUserByToken(token);
    return Right(AuthResult(user: user, accessToken: token));
  } on Exception {
    return Left(ServerFailure("unableToRegister"));
  }
}
