String formatDistance(double? meters) {
  if(meters==null) return "";
  var kms = meters/1000;
  if (kms<1) return [meters.toString(),tr("m")].join(" ");
  if (kms>=1) return [kms.toInt().toString(),tr("km")].join(" ");
  return meters.toString();
}
